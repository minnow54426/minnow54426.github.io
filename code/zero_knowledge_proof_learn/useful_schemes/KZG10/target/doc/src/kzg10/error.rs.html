//! Error types for the KZG10 library
//!
//! This module defines all error types that can occur throughout the library,
//! from setup through commitment, proof generation, and verification.
//!
//! # Design Principles
//!
//! 1. **Security First**: Error messages never leak secret information
//! 2. **Context Rich**: Errors include relevant values for debugging
//! 3. **Type Safe**: Rust's enum system ensures exhaustive error handling
//! 4. **Ergonomic**: Implements std::error::Error and Display for easy integration
//!
//! # Error Categories
//!
//! - **Setup Errors**: Problems with SRS generation or loading
//! - **Operation Errors**: Invalid inputs or state during operations
//! - **Verification Errors**: Proof verification failures
//! - **I/O Errors**: File operations for loading/saving SRS or proofs

/// Result type alias for KZG10 operations
///
/// This is the main return type throughout the library, providing
/// ergonomic error handling with the `?` operator.
///
/// # Example
///
/// ```rust
/// use kzg10::{Error, Result};
///
/// fn validate_degree(degree: usize, max: usize) -> Result<()> {
///     if degree > max {
///         return Err(Error::DegreeTooLarge { max, requested: degree });
///     }
///     Ok(())
/// }
/// ```
pub type Result<T> = std::result::Result<T, Error>;

/// Comprehensive error type for all KZG10 operations
///
/// Each error variant includes context to help diagnose issues
/// without revealing cryptographic secrets.
///
/// # Security Note
///
/// None of these error variants leak information about:
/// - The secret s used in trusted setup
/// - Polynomial coefficients beyond what's necessary
/// - Private keys or other sensitive data
#[derive(Debug, thiserror::Error)]
pub enum Error {
    /// Polynomial degree exceeds SRS maximum capacity
    ///
    /// This occurs when trying to commit to a polynomial with degree
    /// higher than what the SRS was initialized for.
    ///
    /// # When This Occurs
    ///
    /// - During commitment: If poly.degree() > srs.max_degree()
    /// - During proof generation: If evaluation point requires higher degree
    /// - During batch operations: If accumulated degree exceeds capacity
    ///
    /// # Security Implications
    ///
    /// This error reveals the requested and maximum degrees, which is safe
    /// because these are public parameters. No secret information is leaked.
    ///
    /// # Example
    ///
    /// ```rust
    /// use kzg10::Error;
    ///
    /// fn validate_degree() -> Result<(), Error> {
    ///     // SRS initialized for max degree 100
    ///     // Trying to commit to degree 200 polynomial
    ///     let err = Error::DegreeTooLarge { max: 100, requested: 200 };
    ///     Err(err)
    /// }
    /// ```
    #[error("Polynomial degree {requested} exceeds SRS maximum {max}")]
    DegreeTooLarge {
        /// The maximum degree supported by the SRS
        max: usize,
        /// The degree that was requested
        requested: usize,
    },

    /// Degree mismatch between expected and actual polynomial
    ///
    /// This typically indicates a bug in the code or corrupted data.
    ///
    /// # When This Occurs
    ///
    /// - During batch proof verification when degrees don't match
    /// - During polynomial reconstruction
    /// - When deserializing polynomial data with wrong degree field
    ///
    /// # Security Implications
    ///
    /// Degrees are public parameters, so revealing them in error messages
    /// is completely safe. This helps debug integration issues.
    #[error("Degree mismatch: expected {expected}, found {found}")]
    MismatchedDegree {
        /// The degree we expected
        expected: usize,
        /// The degree actually found
        found: usize,
    },

    /// Proof verification failed
    ///
    /// This doesn't necessarily indicate malicious behavior - it can occur
    /// due to transmission errors, data corruption, or honest mistakes.
    ///
    /// # When This Occurs
    ///
    /// - During verification when pairing equation doesn't hold
    /// - When proof structure is valid but cryptographic check fails
    /// - When batch proof verification finds any invalid proof
    ///
    /// # Security Implications
    ///
    /// Verification failure is an expected part of the protocol when
    /// dealing with invalid proofs. The constant-time verification
    /// ensures no timing information leaks about WHY it failed.
    ///
    /// # Why This Error Exists
    ///
    /// Distinguishes between "proof is malformed" and "proof is valid but incorrect"
    /// - InvalidProof: Structural problems (wrong format, missing fields)
    /// - VerificationFailed: Cryptographically invalid (pairing check failed)
    ///
    /// # Example
    ///
    /// ```rust
    /// use kzg10::Error;
    ///
    /// fn verify_proof() -> Result<(), Error> {
    ///     // Verifier computes pairing check and it fails
    ///     Err(Error::VerificationFailed)
    /// }
    /// ```
    #[error("Proof verification failed")]
    VerificationFailed,

    /// Invalid proof structure or format
    ///
    /// This differs from VerificationFailed - it indicates the proof
    /// itself is malformed, not just incorrect.
    ///
    /// # When This Occurs
    ///
    /// - Deserialization fails due to malformed data
    /// - Proof has wrong number of group elements
    /// - Field elements are not in the correct subgroup
    /// - Proof size doesn't match expected format
    ///
    /// # Security Implications
    ///
    /// The error message describes what went wrong structurally but
    /// doesn't reveal any secrets about valid proofs or the SRS.
    ///
    /// # Why This Error Exists
    ///
    /// Helps distinguish between:
    /// 1. Malformed proofs (wrong structure) -> InvalidProof
    /// 2. Well-formed but invalid proofs -> VerificationFailed
    ///
    /// This distinction is important for debugging and for clients
    /// that need to provide feedback to users.
    #[error("Invalid proof: {0}")]
    InvalidProof(String),

    /// Serialization or deserialization error
    ///
    /// Occurs when reading/writing commitments, proofs, or SRS data
    /// in binary or JSON format.
    ///
    /// # When This Occurs
    ///
    /// - Binary deserialization: Wrong byte format, invalid length
    /// - JSON deserialization: Missing fields, wrong types
    /// - Encoding issues: Character encoding problems
    /// - Version mismatches: Old format vs new format
    ///
    /// # Security Implications
    ///
    /// These errors may reveal implementation details about data formats,
    /// but not cryptographic secrets. The actual data isn't included
    /// in error messages for security reasons.
    ///
    /// # Example
    ///
    /// ```rust
    /// use kzg10::Error;
    ///
    /// fn deserialize_proof() -> Result<(), Error> {
    ///     // Failed to deserialize proof from bytes
    ///     Err(Error::SerializationError(
    ///         "Invalid proof length: expected 48 bytes, got 40".to_string()
    ///     ))
    /// }
    /// ```
    #[error("Serialization error: {0}")]
    SerializationError(String),

    /// I/O error during file operations
    ///
    /// Propagates std::io::Error for operations like:
    /// - Loading SRS from disk
    /// - Saving proof batches
    /// - Reading configuration files
    ///
    /// # When This Occurs
    ///
    /// - File not found when loading SRS
    /// - Permission denied when writing proofs
    /// - Disk full when saving large batches
    /// - Network errors when loading from remote storage
    ///
    /// # Security Implications
    ///
    /// I/O errors can leak file paths and other system information,
    /// but this is acceptable because:
    /// 1. File paths are local to the system
    /// 2. No cryptographic secrets are revealed
    /// 3. This information is needed for debugging
    ///
    /// # Why We Use From Trait
    ///
    /// The `#[from]` attribute on std::io::Error allows automatic
    /// conversion from io::Error to our Error type, making error
    /// handling with the `?` operator seamless:
    ///
    /// ```rust
    /// use std::fs::File;
    /// use kzg10::Result;
    ///
    /// fn load_file<T>(path: &str) -> Result<T> {
    ///     // This automatically converts io::Error to Error::IoError
    ///     let _file = File::open(path)?;
    ///     // ... parse file into T
    ///     unimplemented!()
    /// }
    /// ```
    #[error("I/O error: {0}")]
    IoError(#[from] std::io::Error),

    /// Error during SRS setup ceremony
    ///
    /// This is used when the trusted setup process encounters issues.
    ///
    /// # When This Occurs
    ///
    /// - Toxic waste detection fails
    /// - Participant contributions are invalid
    /// - Powers of tau verification fails
    /// - Setup ceremony coordination errors
    ///
    /// # Security Implications
    ///
    /// Setup errors may reveal information about the setup process,
    /// but not the actual toxic waste (secret randomness). This is
    /// safe because the setup process is public and verifiable.
    ///
    /// # Why This Error Exists
    ///
    /// Distinguishes setup-specific errors from general I/O or validation
    /// errors. This is important for setup ceremonies which have unique
    /// failure modes and recovery procedures.
    #[error("Setup error: {0}")]
    SetupError(String),

    /// Invalid input parameters
    ///
    /// Used for validation errors that don't fit other categories.
    ///
    /// # When This Occurs
    ///
    /// - Evaluation point is outside domain
    /// - Empty polynomial list in batch operations
    /// - Invalid parameters passed to public API functions
    /// - Configuration validation failures
    ///
    /// # Security Implications
    ///
    /// May reveal information about expected input formats and ranges,
    /// but this is acceptable because:
    /// 1. These are public API constraints
    /// 2. No secret information is revealed
    /// 3. Helps users debug their integration
    ///
    /// # Example
    ///
    /// ```rust
    /// use kzg10::Error;
    /// use kzg10::Result;
    ///
    /// fn validate_batch<T>(polys: &[T]) -> Result<()> {
    ///     if polys.is_empty() {
    ///         return Err(Error::InvalidInput(
    ///             "Batch cannot be empty".to_string()
    ///         ));
    ///     }
    ///     Ok(())
    /// }
    /// ```
    #[error("Invalid input: {0}")]
    InvalidInput(String),

    /// Invalid SRS structure or verification failed
    ///
    /// This error is used when the SRS (Structured Reference String) is
    /// malformed, fails validation checks, or contains inconsistent data.
    ///
    /// # When This Occurs
    ///
    /// - Loading SRS from file fails validation
    /// - Pairing checks detect inconsistent powers
    /// - SRS has wrong number of G₁ or G₂ elements
    /// - Powers of tau transcript is invalid
    ///
    /// # Security Implications
    ///
    /// This error reveals what validation failed, which is safe because:
    /// 1. The SRS is a public parameter
    /// 2. Validation criteria are public
    /// 3. No secret information about the toxic waste is leaked
    ///
    /// # Why This Error Exists
    ///
    /// Distinguishes SRS-specific validation errors from general I/O errors.
    /// This is important because SRS validation failures indicate potentially
    /// malicious data or corrupted ceremony outputs.
    ///
    /// # Example
    ///
    /// ```rust
    /// use kzg10::Error;
    ///
    /// fn validate_srs() -> Result<(), Error> {
    ///     // Pairing check detected mismatched powers
    ///     Err(Error::InvalidSRS(
    ///         "Pairing check failed: SRS powers are inconsistent".to_string()
    ///     ))
    /// }
    /// ```
    #[error("Invalid SRS: {0}")]
    InvalidSRS(String),

    /// Invalid polynomial degree
    ///
    /// Used when degree validation fails in ways that don't match DegreeTooLarge.
    ///
    /// # When This Occurs
    ///
    /// - Degree is 0 or negative (if using signed types)
    /// - Degree doesn't match expected structure
    /// - Degree validation fails for specific reasons
    ///
    /// # Security Implications
    ///
    /// Degree values are public parameters, so revealing them in error
    /// messages is completely safe.
    #[error("Invalid degree {degree}: {reason}")]
    InvalidDegree {
        /// The invalid degree value
        degree: usize,
        /// Why the degree is invalid
        reason: String,
    },
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_error_display() {
        let err = Error::DegreeTooLarge {
            max: 100,
            requested: 200,
        };
        let msg = format!("{}", err);
        assert!(msg.contains("100"));
        assert!(msg.contains("200"));
        assert!(msg.contains("exceeds SRS max"));
    }

    #[test]
    fn test_error_debug() {
        let err = Error::VerificationFailed;
        let debug_msg = format!("{:?}", err);
        assert!(debug_msg.contains("VerificationFailed"));
    }

    #[test]
    fn test_error_from_io() {
        let io_err = std::io::Error::new(std::io::ErrorKind::NotFound, "file not found");
        let err: Error = io_err.into();
        assert!(matches!(err, Error::IoError(_)));
    }
}
