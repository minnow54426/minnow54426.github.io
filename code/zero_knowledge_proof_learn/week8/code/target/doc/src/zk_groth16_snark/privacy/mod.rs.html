1//! Privacy circuit for range proofs
2//!
3//! This module implements a circuit that proves a value is within a specified
4//! range without revealing the value itself.
5//!
6//! # Example
7//!
8//! ```ignore
9//! let min = 10u64;
10//! let max = 100u64;
11//! let circuit = PrivacyCircuit::new(min, max);
12//! // Later: prove knowledge of value in range [min, max]
13//! ```
14
15use crate::circuit::Groth16Circuit;
16use crate::error::{CircuitError, PrivacyError, Result};
17use ark_bn254::Fr;
18use ark_relations::r1cs::ConstraintSystemRef;
19use serde::{Deserialize, Serialize};
20
21/// Privacy circuit for range proofs
22///
23/// This circuit proves a value is within [min, max] without revealing the value
24#[derive(Debug, Clone, Serialize, Deserialize)]
25pub struct PrivacyCircuit {
26    /// Minimum value of the range
27    pub min: u64,
28    /// Maximum value of the range
29    pub max: u64,
30}
31
32impl PrivacyCircuit {
33    /// Create a new Privacy circuit with the given range
34    ///
35    /// # Arguments
36    ///
37    /// * `min` - Minimum value of the range
38    /// * `max` - Maximum value of the range
39    ///
40    /// # Example
41    ///
42    /// ```
43    /// # use zk_groth16_snark::privacy::PrivacyCircuit;
44    /// let circuit = PrivacyCircuit::new(10, 100);
45    /// assert_eq!(circuit.min, 10);
46    /// assert_eq!(circuit.max, 100);
47    /// ```
48    pub fn new(min: u64, max: u64) -> Self {
49        Self { min, max }
50    }
51}
52
53/// Witness for privacy circuit containing the secret value and bounds
54#[derive(Debug, Clone, Serialize, Deserialize)]
55pub struct PrivacyWitness {
56    /// The secret value to prove is in range
57    pub value: u64,
58    /// Minimum value of the range
59    pub min: u64,
60    /// Maximum value of the range
61    pub max: u64,
62}
63
64impl Groth16Circuit<Fr> for PrivacyCircuit {
65    fn circuit_name() -> &'static str {
66        "privacy"
67    }
68
69    /// Public inputs: the range bounds as u64 values
70    type PublicInputs = (u64, u64);
71
72    /// Private witness: contains the secret value and bounds
73    type Witness = PrivacyWitness;
74
75    fn generate_constraints(cs: ConstraintSystemRef<Fr>, witness: &Self::Witness) -> Result<()> {
76        use ark_r1cs_std::fields::fp::FpVar;
77
78        // Allocate witness variable for the secret value
79        let value_var = FpVar::new_witness(ark_relations::ns!(cs, "value"), || Ok(Fr::from(witness.value)))?;
80
81        // Allocate public inputs for min and max
82        let min_var = FpVar::new_input(ark_relations::ns!(cs, "min"), || Ok(Fr::from(witness.min)))?;
83        let max_var = FpVar::new_input(ark_relations::ns!(cs, "max"), || Ok(Fr::from(witness.max)))?;
84
85        // Enforce range constraint: (value - min) * (max - value) should be non-negative
86        // In R1CS, we enforce the computation is correct
87        let diff_min = value_var.clone() - min_var;
88        let diff_max = max_var - value_var.clone();
89
90        // Compute product
91        let product = diff_min.clone() * diff_max.clone();
92
93        // Allocate witness for product
94        let product_witness = FpVar::new_witness(ark_relations::ns!(cs, "product_witness"), || {
95            let val_min = Fr::from(witness.value) - Fr::from(witness.min);
96            let max_val = Fr::from(witness.max) - Fr::from(witness.value);
97            Ok(val_min * max_val)
98        })?;
99
100        // Enforce the product constraint
101        product.enforce_equal(&product_witness)?;
102
103        Ok(())
104    }
105
106    fn generate_witness(&self) -> Result<Self::Witness> {
107        // For now, return error since we need a value
108        Err(CircuitError::Privacy(PrivacyError::ValueOutOfRange).into())
109    }
110
111    fn public_inputs(witness: &Self::Witness) -> Self::PublicInputs {
112        (witness.min, witness.max)
113    }
114}
115
116impl PrivacyCircuit {
117    /// Generate a witness for a specific value
118    pub fn generate_witness_for_value(&self, value: u64) -> PrivacyWitness {
119        PrivacyWitness {
120            value,
121            min: self.min,
122            max: self.max,
123        }
124    }
125}

127#[cfg(test)]
128mod tests {
129    use super::*;
130
131    #[test]
132    fn test_new() {
133        let circuit = PrivacyCircuit::new(10, 100);
134        assert_eq!(circuit.min, 10);
135        assert_eq!(circuit.max, 100);
136    }
137
138    #[test]
139    fn test_circuit_name() {
140        assert_eq!(PrivacyCircuit::circuit_name(), "privacy");
141    }
142
143    #[test]
144    fn test_zero_bounds() {
145        let circuit = PrivacyCircuit::new(0, 0);
146        assert_eq!(circuit.min, 0);
147        assert_eq!(circuit.max, 0);
148    }
149
150    #[test]
151    fn test_generate_witness_for_value() {
152        let circuit = PrivacyCircuit::new(10, 100);
153        let witness = circuit.generate_witness_for_value(50);
154        assert_eq!(witness.value, 50);
155        assert_eq!(witness.min, 10);
156        assert_eq!(witness.max, 100);
157    }
158
159    #[test]
160    fn test_public_inputs() {
161        let circuit = PrivacyCircuit::new(10, 100);
162        let witness = circuit.generate_witness_for_value(50);
163        let (min, max) = PrivacyCircuit::public_inputs(&witness);
164        assert_eq!(min, 10);
165        assert_eq!(max, 100);
166    }
