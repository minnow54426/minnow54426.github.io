15/// Proving parameters generated during trusted setup

17/// Wrapper around ark-groth16's ProvingKey
18pub type ProvingParams = ProvingKey<Bn254>;

25/// Perform trusted setup for a Groth16 circuit
26///
27/// This generates the proving and verification parameters for a given circuit.
28/// In production, this would require a secure multi-party computation (MPC)
29/// ceremony to ensure the toxic waste is properly destroyed.
30///
31/// # Arguments
32///
33/// * `circuit` - The circuit to generate parameters for
34///
35/// # Returns
36///
37/// A tuple of (ProvingParams, VerificationKey)
38///
39/// # Errors
40///
41/// Returns `SetupError::SetupFailed` if setup cannot be completed
42///
43/// # Example
44///
45/// ```ignore
46/// let circuit = MyCircuit::new();
47/// let (params, vk) = setup(&circuit)?;
48/// ```
49pub fn setup<C: Groth16Circuit<Fr>>(circuit: &C) -> Result<(ProvingParams, VerificationKey)> {
50    // Generate witness for the circuit
51    let witness = circuit
52        .generate_witness()
53        .map_err(|_| Error::Setup(SetupError::SetupFailed))?;
54
55    // Create a constraint synthesizer that will generate constraints
56    // We need to wrap our circuit to implement ConstraintSynthesizer
57    let synthesizer: CircuitSynthesizer<C> = CircuitSynthesizer::new(witness)?;
58
59    // Generate random parameters using Groth16
60    let pk = Groth16::<Bn254>::generate_random_parameters_with_reduction(synthesizer, &mut OsRng)
61        .map_err(|_| Error::Setup(SetupError::SetupFailed))?;
62
63    // Extract the verifying key from the proving key
64    let vk = pk.vk.clone();
65
66    Ok((pk, vk))
67}

69/// Generate a Groth16 proof for a given circuit instance
70///
71/// # Arguments
72///
73/// * `pk` - Proving key from trusted setup
74/// * `witness` - Private witness for the circuit instance
75///
76/// # Returns
77///
78/// A serialized proof as a byte vector
79///
80/// # Errors
81///
82/// Returns `ProveError::ProofCreationFailed` if proof generation fails
83///
84/// # Example
85///
86/// ```ignore
87/// let (pk, _vk) = setup(&circuit)?;
88/// let witness = circuit.generate_witness()?;
89/// let proof = prove(&pk, &witness)?;
90/// ```
91pub fn prove<C>(
92    pk: &ProvingParams,
93    witness: &C::Witness,
94) -> Result<Vec<u8>>
95where
96    C: Groth16Circuit<Fr>,
97{
98    // Create a constraint synthesizer with the witness
99    let synthesizer = CircuitSynthesizer::<C>::new(witness.clone())?;
100
101    // Generate the proof using Groth16
102    let proof = Groth16::<Bn254>::prove(&pk, synthesizer, &mut OsRng)
103        .map_err(|_| Error::Prove(ProveError::ProofCreationFailed))?;
104
105    // Serialize the proof to bytes
106    let mut proof_bytes = vec![];
107    proof
108        .serialize_compressed(&mut proof_bytes)
109        .map_err(|_| Error::Prove(ProveError::ProofCreationFailed))?;
110
111    Ok(proof_bytes)
112}

114/// Verify a Groth16 proof
115///
116/// # Arguments
117///
118/// * `vk` - Verification key from trusted setup
119/// * `public_inputs` - Public inputs for the circuit instance
120/// * `proof` - Serialized proof bytes
121///
122/// # Returns
123///
124/// `true` if the proof is valid, `false` otherwise
125///
126/// # Errors
127///
128/// Returns `VerifyError::ProofVerificationFailed` if verification fails
129///
130/// # Example
131///
132/// ```ignore
133/// let is_valid = verify(&vk, &public_inputs, &proof)?;
134/// assert!(is_valid);
135/// ```
136pub fn verify<C>(
137    vk: &VerificationKey,
138    _public_inputs: &C::PublicInputs,
139    proof: &[u8],
140) -> Result<bool>
141where
142    C: Groth16Circuit<Fr>,
143{
144    // Deserialize the proof
145    let proof_obj = Proof::<Bn254>::deserialize_compressed(proof)
146        .map_err(|_| Error::Verify(VerifyError::InvalidProof))?;
147
148    // Convert public inputs to field elements
149    // For now, we'll use an empty vector since SimpleCircuit doesn't have real constraints
150    let inputs = vec![];
151
152    // Verify the proof
153    let is_valid = Groth16::<Bn254>::verify(&vk, &inputs, &proof_obj)
154        .map_err(|_| Error::Verify(VerifyError::ProofVerificationFailed))?;
155
156    Ok(is_valid)
157}

159/// Adapter struct to convert our Groth16Circuit trait into
160/// ark-relations's ConstraintSynthesizer trait
161struct CircuitSynthesizer<C: Groth16Circuit<Fr>> {
162    witness: C::Witness,
163}
164
165impl<C: Groth16Circuit<Fr>> CircuitSynthesizer<C> {
166    fn new(witness: C::Witness) -> Result<Self> {
167        Ok(Self { witness })
168    }
169}
170
171impl<C: Groth16Circuit<Fr>> ConstraintSynthesizer<Fr> for CircuitSynthesizer<C> {
172    fn generate_constraints(
173        self,
174        cs: ark_relations::r1cs::ConstraintSystemRef<Fr>,
175    ) -> ark_relations::r1cs::Result<()> {
176        // Delegate to our circuit's constraint generation
177        C::generate_constraints(cs, &self.witness)
178            .map_err(|_| ark_relations::r1cs::SynthesisError::Unsatisfiable)?;
179        Ok(())
180    }
181}
