1//! Rust Protocol Basics - Core cryptographic and serialization utilities
2//!
3//! This library provides fundamental building blocks for cryptographic protocols
4//! and secure communication systems. It demonstrates Rust's ownership system,
5//! error handling patterns, and type safety features.
6//!
7//! # Quick Start
8//!
9//! ```rust
10//! use rust_protocol_basics::*;
11//!
12//! // Serialize a struct and hash it
13//! let data = ExampleStruct::new(1, "test".to_string(), vec![1, 2, 3]);
14//! let bytes = data.to_bytes().unwrap();
15//! let hash = sha256(&bytes);
16//! let hash32 = Hash32::new(hash);
17//!
18//! println!("Hash: {}", hash32);
19//! ```
20//!
21//! # Modules
22//!
23//! - [`bytes`] - Hex encoding/decoding and serialization utilities
24//! - [`hash`] - SHA-256 hashing and Merkle tree implementations
25//! - [`types`] - Type-safe wrappers for cryptographic data
26
27// Include all modules

30mod types;

37/// Convenience function to hash data directly into a Hash32
38///
39/// This is a shortcut for Hash32::hash_data() provided for ergonomic reasons.
40///
41/// # Examples
42/// ```
43/// use rust_protocol_basics::hash_to_hash32;
44///
45/// let hash = hash_to_hash32("hello world");
46/// println!("Hash: {}", hash);
47/// ```
48pub fn hash_to_hash32<T: AsRef<[u8]>>(data: T) -> Hash32 {
49    Hash32::hash_data(data)
50}

52#[cfg(test)]
53mod integration_tests {
54    use super::*;
55
56    #[test]
57    fn test_serialize_hash_workflow() {
58        // Demonstrate the complete workflow: serialize -> hash -> display
59        let example = ExampleStruct::new(
60            42,
61            "integration test".to_string(),
62            vec![0x01, 0x02, 0x03, 0x04],
63        );
64
65        // Serialize to bytes
66        let serialized = example.to_bytes().unwrap();
67        assert!(!serialized.is_empty());
68
69        // Hash the bytes
70        let hash_bytes = sha256(&serialized);
71        let hash_wrapper = Hash32::new(hash_bytes);
72
73        // Display as hex
74        let hex_display = format!("{}", hash_wrapper);
75        assert_eq!(hex_display.len(), 64); // SHA-256 is 32 bytes = 64 hex chars
76
77        // Roundtrip: hex string back to Hash32
78        let reconstructed: Hash32 = hex_display.parse().unwrap();
79        assert_eq!(reconstructed, hash_wrapper);
80    }
81
82    #[test]
83    fn test_merkle_tree_with_serialized_data() {
84        // Test Merkle tree with serialized structs
85        let items = [
86            ExampleStruct::new(1, "a".to_string(), vec![1]),
87            ExampleStruct::new(2, "b".to_string(), vec![2]),
88            ExampleStruct::new(3, "c".to_string(), vec![3]),
89        ];
90
91        // Serialize all items
92        let serialized_items: Result<Vec<Vec<u8>>, _> =
93            items.iter().map(|item| item.to_bytes()).collect();
94        let serialized_items = serialized_items.unwrap();
95
96        // Create Merkle tree
97        let tree = MerkleTree::new(&serialized_items);
98        assert_ne!(tree.root(), [0u8; 32]);
99
100        // Verify root is different from simple concatenation
101        let concatenated = serialized_items.concat();
102        let simple_hash = sha256(&concatenated);
103        assert_ne!(tree.root(), simple_hash);
104    }
